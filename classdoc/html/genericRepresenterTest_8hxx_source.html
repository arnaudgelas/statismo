00108                 // as many entries as there are points * dimensions

00110                 VectorType sampleVector = m_representer->SampleToSampleVector(sample);
00111                 if (sampleVector.cols() != Representer::GetDimensions() * domain.GetNumberOfPoints()) {
00112                         std::cout << "the dimension of the sampleVector does not agree with the number of points in the domain (#points * dimensionality)" << std::endl;
00113                 }
00114 
00115 
00116                 unsigned ptNo = 0;
00117                 for (typename DomainType::DomainPointsListType::const_iterator it = domPoints.begin();
00118                                 it != domPoints.end();
00119                                 ++it)
00120                 {
00121                         // since this can take long, we only do it for every 10th point
00122                         if (ptNo % 10 != 0)
00123                                 break;
00124 
00125                         if (m_representer->GetPointIdForPoint(*it) >= domain.GetNumberOfPoints()) {
00126                                 std::cout << "a point in the domain did not evaluate to a valid point it" << std::endl;
